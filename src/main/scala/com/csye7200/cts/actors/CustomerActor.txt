package com.csye7200.cts.actors

import akka.actor.{Actor, ActorLogging}

// CustomerActor
class CustomerActor extends Actor with ActorLogging {
  def receive: Receive = {
    case PurchaseEventTicket(event, ticketSeller) =>
      ticketSeller ! PurchaseTicket(event, self)
    case message: String =>
      log.info(s"Customer received message: $message")
  }
}
